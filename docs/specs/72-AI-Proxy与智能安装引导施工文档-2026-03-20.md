### 2.2 src/index.mjs

```javascript
// AgentShield AI Proxy — Cloudflare Worker
// 保护 MiniMax API Key，验证 Pro License，Rate Limit

const CORS_HEADERS = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Methods': 'POST, OPTIONS',
  'Access-Control-Allow-Headers': 'Content-Type, X-License-ID',
};

export default {
  async fetch(request, env) {
    // CORS preflight
    if (request.method === 'OPTIONS') {
      return new Response(null, { status: 204, headers: CORS_HEADERS });
    }

    // Only POST /v1/chat/completions
    const url = new URL(request.url);
    if (request.method !== 'POST' || !url.pathname.startsWith('/v1/chat/completions')) {
      return jsonResponse(404, { error: 'Not Found' });
    }

    // 1. Extract license ID
    const licenseId = request.headers.get('X-License-ID') || '';
    if (!licenseId) {
      return jsonResponse(401, { error: 'Missing X-License-ID header. Pro license required.' });
    }

    // 2. Verify Pro license
    try {
      const verifyResp = await fetch(`${env.LICENSE_GATEWAY_URL}/client/licenses/verify`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ license_id: licenseId }),
      });
      const verifyData = await verifyResp.json();
      if (!verifyData.valid) {
        return jsonResponse(403, { error: 'Invalid or expired Pro license.', detail: verifyData.reason });
      }
    } catch (err) {
      return jsonResponse(502, { error: 'License verification failed', detail: String(err) });
    }

    // 3. Rate limit (per license, 10 req/60s)
    if (env.PRO_USER_RATE_LIMITER) {
      const { success } = await env.PRO_USER_RATE_LIMITER.limit({ key: licenseId });
      if (!success) {
        return jsonResponse(429, { error: 'Rate limit exceeded. Max 10 requests per minute.' });
      }
    }

    // 4. Forward to MiniMax API
    const apiKey = env.MINIMAX_API_KEY;
    if (!apiKey) {
      return jsonResponse(500, { error: 'AI service not configured' });
    }

    let body;
    try {
      body = await request.json();
    } catch {
      return jsonResponse(400, { error: 'Invalid JSON body' });
    }

    // Force model to prevent abuse
    body.model = body.model || env.MINIMAX_DEFAULT_MODEL;

    const minimaxResp = await fetch(`${env.MINIMAX_BASE_URL}/chat/completions`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(body),
    });

    // 5. Transparent forward response
    const respBody = await minimaxResp.text();
    return new Response(respBody, {
      status: minimaxResp.status,
      headers: {
        'Content-Type': 'application/json',
        ...CORS_HEADERS,
      },
    });
  },
};

function jsonResponse(status, data) {
  return new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json', ...CORS_HEADERS },
  });
}
```

### 4.1 ai_orchestrator.rs — 新增 Pro 内置 AI 调用

> 基于 Tauri v2 官方文档 tauri-plugin-http 的 reqwest 用法

```rust
// 新增函数：通过 Proxy 调 AI（Pro 用户专用）
#[tauri::command]
pub async fn pro_ai_chat(
    license_id: String,
    messages: Vec<serde_json::Value>,
) -> Result<String, String> {
    // 验证 Pro license 本地状态
    crate::commands::license::require_pro("AI 安装引导").await?;

    let proxy_url = std::env::var("AGENTSHIELD_AI_PROXY_URL")
        .unwrap_or_else(|_| "https://agentshield-ai-proxy.pengluailll.workers.dev".to_string());

    let client = reqwest::Client::new();
    let body = serde_json::json!({
        "model": "MiniMax-M2.7",
        "messages": messages,
        "temperature": 0.3,
        "max_tokens": 2000,
    });

    let resp = client
        .post(format!("{}/v1/chat/completions", proxy_url))
        .header("Content-Type", "application/json")
        .header("X-License-ID", &license_id)
        .json(&body)
        .timeout(std::time::Duration::from_secs(60))
        .send()
        .await
        .map_err(|e| format!("AI proxy request failed: {e}"))?;

    if !resp.status().is_success() {
        let status = resp.status().as_u16();
        let text = resp.text().await.unwrap_or_default();
        return Err(format!("AI proxy error {status}: {text}"));
    }

    let json: serde_json::Value = resp
        .json()
        .await
        .map_err(|e| format!("Failed to parse AI response: {e}"))?;

    // 提取 assistant 回复
    let content = json["choices"][0]["message"]["content"]
        .as_str()
        .unwrap_or("AI 没有返回内容")
        .to_string();

    Ok(content)
}
```

## 5. 安全措施总表

| 威胁 | 措施 | 实现层 |
|------|------|--------|
| Key 泄露 | Key 只在 Cloudflare Workers 环境变量中 | Proxy |
| 反编译拿 Key | App 里没有 Key，只有 Proxy URL | App |
| 伪造 License | Proxy 调 license-gateway 验证 | Proxy |
| 滥用/转卖 | Rate Limit 每 license 10次/分钟 | Proxy |
| Key 失效 | `wrangler secret put` 1秒换 Key | Proxy |
| 中间人攻击 | Cloudflare Workers 强制 HTTPS | Proxy |
